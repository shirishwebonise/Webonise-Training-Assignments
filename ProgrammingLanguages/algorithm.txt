
BEGIN
	Read CSV file of restaurants meal info as 'restaurants'
	Accept user's list of items 'userInputItems'

	FOR every restaurant from restaurants, 
	DO
		filter meals to keep only the meals which has an item from userInputItems
		create all possible subsets of the set of meals (using a binary sieve)
		filter the list of the subsets to keep only those which covers all items
		identify lowest cost from the filtered list
	DONE

	Identify which restaurant offers the items at least cost

	IF user's input items not available in any restaurant
	THEN
		DISPLAY "Items you wish are not available in any restaurant"
	END IF
END
